## 方法2 双指针

```javascript
var reverseString = function (s) {
  let l = 0, r = s.length - 1;
  while (l < r) {
    [s[l], s[r]] = [s[r], s[l]];
    l++; r--;
  }
};
```

## 方法3 递归

```javascript
var reverseString = function (s) {
  const swap = (l, r) => {
    if (l >= r) return;
    [s[l], s[r]] = [s[r], s[l]];
    swap(++l, --r);
  }
  swap(0, s.length - 1);
};
```
